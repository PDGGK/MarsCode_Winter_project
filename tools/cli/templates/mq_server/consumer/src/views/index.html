1. Project

list [GET]

url

/project/list

params

名称 类型 描述
psize string 每页条数
pindex string 页码
order string 排序方式
sort string 排序字段
- any 允许搜索应用字段

response

            {
              code: 0,
              data: {
                list: [
                  {
                    id: 'xxx',
                    name: 'xxx',
                    leader: 'xxx',
                    desc: 'xxx',
                    ctime: 'xxx'
                  }
                ],
                total: 1
              },
              msg: 'success'
            }

3. Log

list [GET]

url

/log/list

params

名称 类型 描述
psize string 每页条数
pindex string 页码
order string 排序方式
sort string 排序字段
- any 允许搜索应用字段

response

            {
              code: 0,
              data: {
                list: [
                  {
                    id: 'xxx',
                    type: 'xxx', // 类型
                    sub_type: 'xxx', // 子类型
                    ascription: 'xxx', // 归属应用
                    data: 'xxx', // 详细信息
                    otime: 'xxx' // 发生时间
                  }
                ],
                total: 1
              },
              msg: 'success'
            }

detail [GET]

url

/log/detail

params

名称 类型 描述
id string 日志id

response

            {
              code: 0,
              data: {
                path: 'xxx', // 页面路径
                type: 'xxx', // 类型
                sub_type: 'xxx', // 子类型
                ascription_name: 'xxx', // 归属应用名称
                ctime: 'xxx', // 发生时间
                page_title: 'xxx', // 页面标题
                user_agent: 'xxx', // 浏览器userAgent
                data: 'xxx', // 详细数据
                breadcrumb: [] // 面包屑
              },
              msg: 'success'
            }

4. SourceMap

search [GET]

url

/sourcemap/search

params

名称 类型 描述
lineno number 行数
colno number 列数
filename string 文件名称
app_name string 应用名称

response

            {
              code: 0,
              data: {
                column: '',
                line: '',
                source: '',
                code: ''
              },
              msg: 'success'
            }
